    import mysql from 'mysql2/promise';
import 'dotenv/config';

const pool = mysql.createPool({
  host: process.env.DB_HOST,
  port: Number(process.env.DB_PORT ?? 3306),
  user: process.env.DB_USER,
  password: process.env.DB_PASSWORD,
  database: process.env.DB_NAME,
  waitForConnections: true,
  connectionLimit: 10,
  queueLimit: 0,
  namedPlaceholders: true
});

// Opcional: comprovació inicial de connexió
(async () => {
  try {
    const conn = await pool.getConnection();
    await conn.ping();
    console.log('✅ Connexió a MySQL correcta');
    conn.release();
  } catch (err) {
    console.error('❌ Error connectant a MySQL:', err.message);
  }
})();

export default pool;

    import express from 'express';
import 'dotenv/config';
import pool from './db.js';

const app = express();
app.use(express.json());

// GET /llibres -> retorna tots els llibres
app.get('/llibres', async (req, res, next) => {
  try {
    const [rows] = await pool.query(
      'SELECT id, titol, autor, any_publicacio, isbn FROM llibres ORDER BY id DESC'
    );
    res.json({ ok: true, data: rows });
  } catch (err) {
    next(err);
  }
});

// (Opcional) GET /llibres/:id -> un llibre concret
app.get('/llibres/:id', async (req, res, next) => {
  try {
    const [rows] = await pool.query(
      'SELECT id, titol, autor, any_publicacio, isbn FROM llibres WHERE id = ? LIMIT 1',
      [req.params.id]
    );
    if (rows.length === 0) return res.status(404).json({ ok: false, error: 'No trobat' });
    res.json({ ok: true, data: rows[0] });
  } catch (err) {
    next(err);
  }
});

// Middleware d'errors (respostes coherents)
app.use((err, req, res, next) => {
  console.error(err);
  res
    .status(500)
    .json({ ok: false, error: 'Error intern del servidor', detail: process.env.NODE_ENV === 'production' ? undefined : err.message });
});

const PORT = Number(process.env.PORT ?? 3000);
app.listen(PORT, () => {
  console.log(`🚀 API escoltant a http://localhost:${PORT}`);
});
